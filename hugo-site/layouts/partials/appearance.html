{{- /* Light/dark theming. Light vars are the default; dark applies via the OS preference (when mode=auto) and can be forced with the header toggle, which stores the choice in localStorage as data-theme on . */ -}} {{- $a := .Site.Params.appearance -}} {{- $font := "system" -}}{{- with $a.font }}{{ $font = . }}{{ end -}} {{- $stacks := dict "system" "-apple-system,BlinkMacSystemFont,'Segoe UI',Roboto,Helvetica,Arial,sans-serif" "sans" "Helvetica,Arial,'Helvetica Neue',sans-serif" "serif" "Georgia,'Times New Roman',Times,serif" "mono" "ui-monospace,SFMono-Regular,Menlo,Consolas,monospace" "rounded" "'Avenir Next','Segoe UI',system-ui,sans-serif" -}} {{- $stack := index $stacks $font -}}{{- if not $stack }}{{ $stack = index $stacks "system" }}{{ end -}} {{- $mode := $a.mode | default "auto" -}} {{- $la := $a.lightaccent | default "#111111" -}}{{- $lb := $a.lightbg | default "#ffffff" -}}{{- $lt := $a.lighttext | default "#222426" -}} {{- $da := $a.darkaccent | default "#8ab4ff" -}}{{- $db := $a.darkbg | default "#15171a" -}}{{- $dt := $a.darktext | default "#e6e8eb" -}} {{- /* Optional custom Google Fonts. Body font overrides the stack; heading font drives the site name + titles via --heading-font. */ -}} {{- $headUrl := $a.headingfonturl -}}{{- $headFam := $a.headingfontfamily -}} {{- $bodyUrl := $a.bodyfonturl -}}{{- $bodyFam := $a.bodyfontfamily -}} {{- $bodyStack := $stack -}}{{- with $bodyFam }}{{ $bodyStack = printf "'%s', %s" . $stack }}{{ end -}} {{- $headStack := "var(--user-font)" -}}{{- with $headFam }}{{ $headStack = printf "'%s', %s" . $bodyStack }}{{ end -}} {{ if or $headUrl $bodyUrl }}{{ end }} {{ with $headUrl }}{{ end }} {{ if and $bodyUrl (ne $bodyUrl $headUrl) }}{{ end }}
